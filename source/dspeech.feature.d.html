11 /**
12    Computes a magnitude 2-D spectrogram from a time-domain 1-D signal
13 
14 Params:
15     windowFun = window function
16     xs = 1-D slice signal
17     nperseg = (default 256) short-time frame width for each FFT segment
18     noverlap = (default nperseg / 2) short-time frame overlapped length for each FFT segment
19 
20 Returns: a magnitude 2D spectrogram
21 */
22 auto spectrogram(alias windowFun=hann, S)(S xs, size_t nperseg = 256, size_t noverlap = 128)
23 {
24     import numir.signal : stft;
25     import mir.math.common : sqrt;
26     import mir.ndslice.topology : map;
27     import mir.ndslice : transposed;
28 
29     auto stfs = stft!windowFun(xs, nperseg, noverlap);
30     auto upper = stfs.transposed[nperseg / 2 .. $];
31     auto mag = upper.map!(a => sqrt(a.re * a.re + a.im * a.im));
32     return mag;
33 }

62 /**
63 Computes Mel-scale filterbank matrix
64 
65 Params:
66     nFreq: the number of FFT bins
67     nMel: the number of filterbank bins
68     sampleFreq: sampling frequency of FFT signal
69     lowFreq: lowest frequency threshold of the filterbank
70     highFreq: highest (relative from Nyquist freq) frequency threshold of the filterbank
71 
72 TODO:
73     make this function @nogc
74  */
75 @safe nothrow pure melMatrix(Dtype = double)(size_t nFreq, size_t nMel, Dtype sampleFreq, Dtype lowFreq= 20, Dtype highFreq = 0)
76 do
77 {
78     import mir.ndslice : iota, as, sliced, map;
79     import numir : zeros;
80     const nyquist = 0.5 * sampleFreq;
81     highFreq += nyquist;
82     const lowMel = lowFreq.melScale;
83     const highMel = highFreq.melScale;
84     const deltaMel = (highMel - lowMel) / nMel;
85     const deltaFreq = nyquist / nFreq;
86     const binsMel = iota(nMel + 2).as!Dtype * deltaMel + lowMel;
87 
88     return iota(nFreq, nMel).map!(
89         (i) {
90             const iFreq = i / nMel;
91             const iMel = i % nMel;
92             const leftMel = binsMel[iMel];
93             const centerMel = binsMel[iMel + 1];
94             const rightMel = binsMel[iMel + 2];
95             const mel = melScale(deltaFreq * iFreq);
96             if (mel >= rightMel)
97             {
98                 assert(iFreq > 0, "too large nMel you set");
99                 return cast(Dtype) 0;
100             }
101             if (mel > leftMel)
102             {
103                 return (mel <= centerMel ? mel - leftMel : rightMel - mel) / deltaMel;
104             }
105             return cast(Dtype) 0;
106         });
107 }
