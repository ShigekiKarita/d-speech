11 /**
12    Computes a magnitude 2-D spectrogram from a time-domain 1-D signal
13 
14 Params:
15     windowFun = window function
16     xs = 1-D slice signal
17     nperseg = (default 256) short-time frame width for each FFT segment
18     noverlap = (default nperseg / 2) short-time frame overlapped length for each FFT segment
19 
20 Returns: a magnitude 2D spectrogram
21 */
22 auto spectrogram(alias windowFun=hann, S)(S xs, size_t nperseg = 256, size_t noverlap = 128)
23 {
24     import numir.signal : stft;
25     import mir.math.common : sqrt;
26     import mir.ndslice.topology : map;
27     import mir.ndslice : transposed;
28 
29     auto stfs = stft!windowFun(xs, nperseg, noverlap);
30     auto upper = stfs.transposed[nperseg / 2 .. $];
31     auto mag = upper.map!(a => sqrt(a.re * a.re + a.im * a.im));
32     return mag;
33 }

62 /**
63 Computes Mel-scale filterbank matrix
64 
65 Params:
66     nFreq: the number of FFT bins
67     nMel: the number of filterbank bins
68     sampleFreq: sampling frequency of FFT signal
69     lowFreq: lowest frequency threshold of the filterbank
70     highFreq: highest (relative from Nyquist freq) frequency threshold of the filterbank
71 
72 TODO:
73     make this function @nogc
74  */
75 @safe nothrow pure melMatrix(Dtype = double)(size_t nFreq, size_t nMel, Dtype sampleFreq, Dtype lowFreq= 20, Dtype highFreq = 0)
76 {
77     import mir.ndslice : iota, as, sliced, map;
78     import numir : zeros;
79     const nyquist = 0.5 * sampleFreq;
80     highFreq += nyquist;
81     const lowMel = lowFreq.melScale;
82     const highMel = highFreq.melScale;
83     const deltaMel = (highMel - lowMel) / nMel;
84     const deltaFreq = nyquist / nFreq;
85     const binsMel = iota(nMel + 2).as!Dtype * deltaMel + lowMel;
86 
87     return iota(nFreq, nMel).map!(
88         (i) {
89             const iFreq = i / nMel;
90             const iMel = i % nMel;
91             const leftMel = binsMel[iMel];
92             const centerMel = binsMel[iMel + 1];
93             const rightMel = binsMel[iMel + 2];
94             const mel = melScale(deltaFreq * iFreq);
95             if (mel >= rightMel)
96             {
97                 assert(iFreq > 0, "too large nMel you set");
98                 return cast(Dtype) 0;
99             }
100             if (mel > leftMel)
101             {
102                 return (mel <= centerMel ? mel - leftMel : rightMel - mel) / deltaMel;
103             }
104             return cast(Dtype) 0;
105         });
106 }

119 /**
120 Computes type-II DCT on 1-d signal:
121 
122 $(MATH y[k] = 2 f \sum_{n=0}^{N-1} x[n] \cos \pi k \frac{2n + 1}{2 N} ),
123 
124 where the scaling factor f = sqrt(1 / 4 N) if k = 0,  f = sqrt(1 / 2 N) otherwise,
125 
126 Params:
127     xs = input array
128 
129 Returns:
130     DCT transformed 1-d array
131 
132 See_also: https://docs.scipy.org/doc/scipy/reference/generated/scipy.fftpack.dct.html
133  */
134 auto dct(alias sumKind = "precise", S)(S xs)
135 {
136     import std.math : PI;
137     import mir.math.common;
138     import mir.math.sum;
139     import mir.primitives : DimensionCount;
140     static assert(DimensionCount!S == 1);
141     import mir.ndslice;
142     const N = xs.length;
143     const f0 = 2 * sqrt(0.25 / N);
144     const f = 2 * sqrt(0.5 / N);
145     return  iota(N).map!(k => (k == 0 ? f0 : f) * iota(N).map!(n => xs[n] * cos(PI * k * (2 * n + 1) / (2 * N))).sum!sumKind);
146 }
